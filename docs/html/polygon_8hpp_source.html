254struct PolygonProjection
255{
256 float minProj;
257 float maxProj;
258
264 PolygonProjection(const Polygon &poly, const Point &projAxis)
265 {
266 // Point normAxis = projAxis.normalizeVector();
267
268 minProj = poly.vertices[0].dotProduct(projAxis);
269 maxProj = minProj;
270
271 for (int i = 1; i < poly.vertexCount; i++)
272 {
273 float testedProj = poly.vertices[i].dotProduct(projAxis);
274
275 if (testedProj < minProj)
276 minProj = testedProj;
277 if (testedProj > maxProj)
278 maxProj = testedProj;
279 }
280 }
281
287 bool doProjectionOverlap(const PolygonProjection &other)
288 {
289 return !(maxProj < other.minProj || other.maxProj < minProj);
290 }
